\section{Installation Guide}

\subsection{Local Installation}

\textbf{Step 1: Install Python}
\begin{verbatim}
# Verify Python installation
python --version  # Should be 3.10 or higher
\end{verbatim}

\textbf{Step 2: Clone Repository}
\begin{verbatim}
git clone https://github.com/username/shadeai.git
cd shadeai
\end{verbatim}

\textbf{Step 3: Create Virtual Environment}
\begin{verbatim}
python -m venv .venv
# Windows:
.venv\Scripts\activate
# macOS/Linux:
source .venv/bin/activate
\end{verbatim}

\textbf{Step 4: Install Dependencies}
\begin{verbatim}
pip install -r requirements.txt
\end{verbatim}

\textbf{Step 5: Test Installation}
\begin{verbatim}
python shadeai_cli.py test1.jpg
\end{verbatim}

\subsection{Docker Installation}

\textbf{Step 1: Install Docker}
\begin{itemize}
    \item Download from https://www.docker.com/
    \item Follow installation instructions for your OS
\end{itemize}

\textbf{Step 2: Build and Run}
\begin{verbatim}
docker-compose up --build
\end{verbatim}

\textbf{Step 3: Access Applications}
\begin{itemize}
    \item Frontend: http://localhost:8501
    \item Backend API: http://localhost:8000
    \item API Documentation: http://localhost:8000/docs
\end{itemize}

\section{Usage Examples}

\subsection{Command-Line Interface}

\textbf{Basic Usage:}
\begin{verbatim}
python shadeai_cli.py image.jpg
\end{verbatim}

\textbf{Batch Processing:}
\begin{verbatim}
for image in *.jpg; do
    python shadeai_cli.py "$image"
done
\end{verbatim}

\subsection{Python API}

\textbf{Example Code:}
\begin{verbatim}
import requests

# Upload image
with open('photo.jpg', 'rb') as f:
    response = requests.post(
        'http://localhost:8000/api/v1/analyze',
        files={'file': f}
    )

# Parse results
result = response.json()
print(f"Skin Tone: {result['skin_tone']['label']}")
print(f"Undertone: {result['undertone']['label']}")

# Get recommendations
for rec in result['recommendations']:
    print(f"{rec['brand']} {rec['shade']}: ΔE={rec['delta_e']}")
\end{verbatim}

\subsection{Web Application}

\textbf{Starting the Application:}
\begin{verbatim}
# Terminal 1: Start backend
uvicorn backend.main:app --reload --port 8000

# Terminal 2: Start frontend
streamlit run streamlit_app.py
\end{verbatim}

\textbf{Using the Application:}
\begin{enumerate}
    \item Open http://localhost:8501 in browser
    \item Upload photo or use camera
    \item Review quality assessment
    \item View skin tone analysis
    \item Browse shade recommendations
    \item Download results if needed
\end{enumerate}

\section{Troubleshooting}

\subsection{Common Issues}

\textbf{Issue: Face not detected}
\begin{itemize}
    \item Solution: Ensure face is frontal and well-lit
    \item Solution: Try with relaxed detection parameters
    \item Solution: Check photo quality score
\end{itemize}

\textbf{Issue: Poor accuracy}
\begin{itemize}
    \item Solution: Improve lighting conditions
    \item Solution: Remove heavy makeup or filters
    \item Solution: Use higher resolution image
    \item Solution: Follow photo guidelines
\end{itemize}

\textbf{Issue: Slow processing}
\begin{itemize}
    \item Solution: Reduce image resolution
    \item Solution: Close other applications
    \item Solution: Check CPU usage
\end{itemize}

\textbf{Issue: Installation errors}
\begin{itemize}
    \item Solution: Update pip: \texttt{pip install --upgrade pip}
    \item Solution: Install build tools
    \item Solution: Use virtual environment
\end{itemize}
