\subsection{Assumptions and Dependencies}

\subsubsection{Assumptions}

\begin{enumerate}
    \item \textbf{Photo Quality:}
    \begin{itemize}
        \item Input images are reasonably well-lit
        \item Faces are visible and frontal
        \item No extreme filters or heavy makeup
        \item Resolution ≥ 640×480 pixels
    \end{itemize}
    
    \item \textbf{Color Accuracy:}
    \begin{itemize}
        \item Camera captures colors reasonably accurately
        \item No extreme color casts (correctable by white balance)
        \item sRGB color space for input images
    \end{itemize}
    
    \item \textbf{Foundation Database:}
    \begin{itemize}
        \item CIELAB values are representative of actual products
        \item Shade formulations remain consistent
        \item Database is periodically updated
    \end{itemize}
    
    \item \textbf{User Behavior:}
    \begin{itemize}
        \item Users follow photo guidelines
        \item Users understand limitations
        \item Users verify recommendations before purchase
    \end{itemize}
\end{enumerate}

\subsubsection{Dependencies}

\begin{enumerate}
    \item \textbf{External Libraries:}
    \begin{itemize}
        \item OpenCV for computer vision operations
        \item NumPy for numerical computations
        \item FastAPI and Streamlit for interfaces
    \end{itemize}
    
    \item \textbf{Pre-trained Models:}
    \begin{itemize}
        \item Haar Cascade classifier (included in OpenCV)
        \item No custom training required
    \end{itemize}
    
    \item \textbf{Data Files:}
    \begin{itemize}
        \item foundation\_shades.json (187 shades)
        \item Example images for testing
    \end{itemize}
    
    \item \textbf{Network:}
    \begin{itemize}
        \item Internet connection for web app deployment
        \item Local network for API access
    \end{itemize}
\end{enumerate}

\subsection{Constraints (If Applicable)}

\begin{enumerate}
    \item \textbf{Technical Constraints:}
    \begin{itemize}
        \item Single-face analysis only
        \item No real-time video processing
        \item Limited to static images
        \item CPU-only processing (no GPU acceleration)
    \end{itemize}
    
    \item \textbf{Data Constraints:}
    \begin{itemize}
        \item Foundation database limited to 187 shades
        \item CIELAB values estimated from product images
        \item No direct spectrophotometer measurements
    \end{itemize}
    
    \item \textbf{Accuracy Constraints:}
    \begin{itemize}
        \item Dependent on photo quality
        \item Affected by lighting conditions
        \item Undertone detection uses heuristics
        \item No deep learning models (by design)
    \end{itemize}
    
    \item \textbf{Deployment Constraints:}
    \begin{itemize}
        \item Requires Python environment
        \item Web app needs server hosting
        \item API requires network access
    \end{itemize}
\end{enumerate}

\subsubsection{Edge Cases and Failure Modes}

\textbf{Tested Edge Cases:}
\begin{enumerate}
    \item \textbf{Poor Lighting:}
    \begin{itemize}
        \item Result: Quality checker detects issue
        \item Recommendation: User guidance to retake photo
        \item Accuracy: Reduced but still functional
    \end{itemize}
    
    \item \textbf{Heavy Makeup:}
    \begin{itemize}
        \item Result: May affect undertone detection
        \item Recommendation: System warns about makeup
        \item Accuracy: Tone detection remains accurate
    \end{itemize}
    
    \item \textbf{Side Profile:}
    \begin{itemize}
        \item Result: Face detection may fail
        \item Recommendation: Fallback to full image
        \item Accuracy: Reduced accuracy
    \end{itemize}
    
    \item \textbf{Multiple Faces:}
    \begin{itemize}
        \item Result: Largest face selected
        \item Recommendation: Single-person photos preferred
        \item Accuracy: Normal for selected face
    \end{itemize}
\end{enumerate}

\section{Error Analysis}

\subsection{Sources of Error}

\begin{enumerate}
    \item \textbf{Photo Quality (30-40\%):}
    \begin{itemize}
        \item Poor lighting
        \item Color casts
        \item Low resolution
        \item Blur
    \end{itemize}
    
    \item \textbf{Database Limitations (20-30\%):}
    \begin{itemize}
        \item Estimated CIELAB values
        \item Limited shade coverage in some ranges
        \item Undertone variations
    \end{itemize}
    
    \item \textbf{Segmentation Errors (15-20\%):}
    \begin{itemize}
        \item Inclusion of non-skin pixels
        \item Exclusion of skin pixels
        \item Shadows and highlights
    \end{itemize}
    
    \item \textbf{Individual Variation (10-15\%):}
    \begin{itemize}
        \item Skin tone variations across face
        \item Seasonal changes
        \item Makeup effects
    \end{itemize}
\end{enumerate}
